By using the given Patient Appointment show/noshow excel data, This project plots the overall show and noshow count¶

This project then aims to determine, how below factors influence the show/noshow of the patient 1) Gender of the patient 2) SMS Remainder message sent to the patient 3) Age of the patient¶

This project also studies how the show percentage of the patients varied with the Appointment day (Time period)¶

#Define a function to categorise the data based on Age

def classifier(row):
    if row["Age"] >= 0 and row["Age"] <= 12:
        return "Child ( age  0 - 12)"
    elif row["Age"] >= 13 and row["Age"] <= 20:
        return "Teen ( age  13 -20 )"
    elif row["Age"] >= 21 and row["Age"] <= 60:
        return "Adult ( age 21 - 60)"
    else:
        return "Senior ( age > 60 years)"

    # Read the cleanedup data
    
df_data = pd.read_csv('noshowappointments_clean.csv')


#Introduce a new column by applying the age classifier function
df_data["Agegroup"] = df_data.apply(classifier, axis=1)

df_classify = df_data.groupby(['Agegroup','No-show'])['PatientId'].count()
df_pct = df_classify.groupby(['Agegroup']).apply(lambda x:100 * (x/float(x.sum())))
df_pct.to_csv('noshowpercentage.csv',encoding="utf8",index=True)
df_pctvalue = pd.read_csv('noshowpercentage.csv',header=None)
df_pctvalue.columns = ["Agegroup", "Noshowindicator", "Percentage"]
df_noshow = df_pctvalue.query('Noshowindicator == "Yes" ')
df_show = df_pctvalue.query('Noshowindicator == "No" ')
df_show=df_show.rename(columns = {'Percentage':'Showpercentage'})
df_noshow=df_noshow.rename(columns = {'Percentage':'NoShowpercentage'})

#Data merge to format the data
df_age_plot = df_show.merge(df_noshow,on='Agegroup')

cols = [1,3]
